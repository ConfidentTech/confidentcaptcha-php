  82. /**
  83.  * Policy factory for Confident CAPTCHA policies
  84.  *
  85.  * When recreating a policy with the help of the persistence engine, this
  86.  * class can be used to create or restore a policy by class name.
  87.  *
  88.  * @package confidentcaptcha-php
  89.  */
  90. class CCAP_PolicyFactory
  91. {
  92.     /**
  93.      * Construction parameters for a policy
  94.      * @var array 
  95.      */
  96.     static protected $policies = array();
  97.     
  98.     /**
  99.      * Static class has been initialized
  100.      * @var boolean 
  101.      */
  102.     static protected $initialized = FALSE;
  103.  
  104.     /**
  105.      * Initialize the Policy Factory
  106.      */
  107.     static protected function init()
  108.     {
  109.         if (!CCap_PolicyFactory::$initialized) {
  110.             CCap_PolicyFactory::$initialized = NULL;
  111.             CCap_PolicyFactory::add('CCAP_ProductionFailOpen',
  112.                 'confidentcaptcha/ccap_prod_open_policy.php');
  113.             CCap_PolicyFactory::add('CCAP_ProductionFailClosed',
  114.                 'confidentcaptcha/ccap_prod_closed_policy.php');
  115.             CCap_PolicyFactory::add('CCAP_DevelopmentPolicy',
  116.                 'confidentcaptcha/ccap_dev_policy.php');
  117.             CCap_PolicyFactory::$initialized = TRUE;
  118.         }
  119.     }
  120.  
  121.     /**
  122.      * Does the factory know about a policy?
  123.      *
  124.      * @param string $policy_name The name of the policy
  125.      * @return boolean TRUE if known, FALSE if unknown
  126.      */
  127.     static public function known($policy_name)
  128.     {
  129.         CCap_PolicyFactory::init();
  130.         return (isset(CCap_PolicyFactory::$policies[$policy_name]));
  131.     }
  132.     
  133.     /**
  134.      * Teach the factory about a new policy
  135.      *
  136.      * @param string $policy_name The name of the policy class
  137.      * @param string $policy_include The name of the policy include file
  138.      */
  139.     static public function add($policy_name, $policy_include=NULL)
  140.     {
  141.         // Don't call init if we're in init
  142.         if (!is_null(CCap_PolicyFactory::$initialized)) {
  143.             CCap_PolicyFactory::init();
  144.         }
  145.         CCap_PolicyFactory::$policies[$policy_name] = $policy_include;
  146.     }
  147.  
  148.     /**
  149.      * Create a new policy by name
  150.      *
  151.      * @param string $policy_name The name of the policy to create
  152.      * @param CCAP_Api $api The API class to pass to the policy
  153.      * @param CCAP_Persistance $persist The persistance class for the policy
  154.      * @return CCAP_Policy A newly created policy
  155.      */
  156.     static public function create($policy_name, $api, $persist)
  157.     {
  158.         CCap_PolicyFactory::init();
  159.         if (CCap_PolicyFactory::known($policy_name)) {
  160.             $policy_include = CCap_PolicyFactory::$policies[$policy_name];
  161.             if ($policy_include) include_once($policy_include);
  162.             return new $policy_name($api, $persist);
  163.         } else {
  164.             die("CCAP_PolicyFactory: Unknown policy '$policy_name'");
  165.             return NULL;
  166.         }
  167.     }
  168.  
  169.     /**
  170.      * Recreate a policy with the help of the persistence framework.
  171.      *
  172.      * If the persistance framework doesn't have a stored policy name, then
  173.      * the $fallback_policy_name is used to create the policy.  If this is
  174.      * unset, die() is called.
  175.      *
  176.      * @param CCAP_Persist $persist The persistence engine to query
  177.      * @param CCAP_API $api The API class to pass to the policy
  178.      * @param string $fallback_policy_name The fallback new policy to create
  179.      * @return CCAP_Policy The restored policy or a new fallback policy
  180.      */
  181.     static public function restore($persist, $api, $fallback_policy_name=NULL)
  182.     {
  183.         CCap_PolicyFactory::init();
  184.         $policy_name = $persist->policy_name();
  185.         if ($policy_name) {
  186.             $policy = CCap_PolicyFactory::create($policy_name, $api,
  187.                 $persist);
  188.             $persist->load($policy);
  189.             return $policy;
  190.         } else {
  191.             $policy = CCap_PolicyFactory::create($fallback_policy_name, $api,
  192.                 $persist);
  193.             $policy->reset();
  194.             return $policy;
  195.         }
  196.     }
  197. }
