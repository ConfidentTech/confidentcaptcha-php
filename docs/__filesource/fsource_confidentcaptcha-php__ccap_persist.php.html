  82. /**
  83.  * Persistence interface for Confident CAPTCHA Policies

  88.  * forms inputs.  This class provides the interface for policies to store or
  89.  * or load information from the chosen persistence strategy.

  173. /**
  174.  * Session-based persistence for Confident CAPTCHA Policies
  175.  *
  176.  * Use PHP's $_SESSION to store Confident CAPTCHA state
  177.  *
  178.  * @package confidentcaptcha-php
  179.  */
  180. class CCAP_PersistSession extends CCAP_Persistence
  181. {   
  182.     /**
  183.      * Load state from session
  184.      *
  185.      * @var CCAP_Policy $policy The policy instance to initialize
  186.      */
  187.     public function load(&$policy)
  188.     {
  189.         // Start the session if it hasn't been started yet
  190.         if (!session_id()) session_start();
  191.         
  192.         // Load state from session
  193.         $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  194.         if (isset($_SESSION) and isset($_SESSION[$state_key])) {
  195.             foreach($_SESSION[$state_key] as $key => $value) {
  196.                 if (is_scalar($value) or is_array($value)) {
  197.                     $policy->$key = $value;
  198.                 }
  199.             }
  200.         }
  201.     }
  202.  
  203.     /**
  204.      * Get stored policy name
  205.      * @return string The policy class name
  206.      */
  207.     public function policy_name()
  208.     {
  209.         // Start the session if it hasn't been started yet
  210.         if (!session_id()) session_start();
  211.         
  212.         $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  213.         if (isset($_SESSION) and isset($_SESSION[$name_key])) {
  214.             return $_SESSION[$name_key];
  215.         } else {
  216.             return NULL;
  217.         }
  218.     }
  219.  
  220.     /**
  221.      * Save state to session
  222.      *
  223.      * @var CCAP_Policy $policy The policy instance to save
  224.      */
  225.     public function save(&$policy)
  226.     {
  227.         // Start the session if it hasn't been started yet
  228.         if (!session_id()) session_start();
  229.  
  230.         // Save state to session
  231.         $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  232.         $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  233.         if (isset($_SESSION)) {
  234.             // Store policy name
  235.             $_SESSION[$name_key] = get_class($policy);
  236.             
  237.             // Store policy
  238.             $vars = get_object_vars($policy);
  239.             $out_vars = Array();
  240.             foreach($vars as $key => $value) {
  241.                 if (is_scalar($value) or is_array($value)) {
  242.                     $out_vars[$key] = $value;
  243.                 }
  244.             }
  245.             $_SESSION[$state_key] = $out_vars;
  246.         }
  247.     }
  248.  
  249.     /**
  250.      * Reset state in session
  251.      */
  252.      public function reset(&$policy)
  253.      {
  254.          // Start the session if it hasn't been started yet
  255.          if (!session_id()) session_start();
  256.  
  257.          // Unset state
  258.          $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  259.          $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  260.          if (isset($_SESSION)) {
  261.              unset($_SESSION[$state_key]);
  262.              unset($_SESSION[$name_key]);
  263.          }
  264.      }
  265. }
