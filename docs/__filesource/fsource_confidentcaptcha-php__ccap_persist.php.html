  82. /**
  83.  * Persistence interface for Confident CAPTCHA Policies

  88.  * forms inputs.  This class provides the interface for policies to store or
  89.  * or load information from the chosen persistence strategy.

  183. /**
  184.  * Session-based persistence for Confident CAPTCHA Policies
  185.  *
  186.  * Uses PHP's $_SESSION to store Confident CAPTCHA state.  All functions
  187.  * require a session.  Either call session_start() at the top of a PHP
  188.  * page that will use CAPTCHA, or, better yet, call
  189.  * CCAP_Policy->start_captcha_page() at the top of the PHP page,
  190.  * before any output or headers are sent.
  191.  *
  192.  * @package confidentcaptcha-php
  193.  */
  194. class CCAP_PersistSession extends CCAP_Persistence
  195. {   
  196.     /**
  197.      * Load state from session
  198.      * @var CCAP_Policy $policy The policy instance to initialize
  199.      */
  200.     public function load(&$policy)
  201.     {
  202.         // Start the session if it hasn't been started yet
  203.         if (!session_id()) session_start();
  204.         
  205.         // Load state from session
  206.         $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  207.         if (isset($_SESSION) and isset($_SESSION[$state_key])) {
  208.             foreach($_SESSION[$state_key] as $key => $value) {
  209.                 if (is_scalar($value) or is_array($value)) {
  210.                     $policy->$key = $value;
  211.                 }
  212.             }
  213.         }
  214.     }
  215.  
  216.     /**
  217.      * Get the stored policy name from the SESSION
  218.      * @return string The policy class name, or NULL if none stored
  219.      */
  220.     public function policy_name()
  221.     {
  222.         // Start the session if it hasn't been started yet
  223.         if (!session_id()) session_start();
  224.         
  225.         $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  226.         if (isset($_SESSION) and isset($_SESSION[$name_key])) {
  227.             return $_SESSION[$name_key];
  228.         } else {
  229.             return NULL;
  230.         }
  231.     }
  232.  
  233.     /**
  234.      * Save state to session
  235.      *
  236.      * @var CCAP_Policy $policy The policy instance to save
  237.      */
  238.     public function save(&$policy)
  239.     {
  240.         // Start the session if it hasn't been started yet
  241.         if (!session_id()) session_start();
  242.  
  243.         // Save state to session
  244.         $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  245.         $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  246.         if (isset($_SESSION)) {
  247.             // Store policy name
  248.             $_SESSION[$name_key] = get_class($policy);
  249.             
  250.             // Store policy
  251.             $vars = get_object_vars($policy);
  252.             $out_vars = Array();
  253.             foreach($vars as $key => $value) {
  254.                 if (is_scalar($value) or is_array($value)) {
  255.                     $out_vars[$key] = $value;
  256.                 }
  257.             }
  258.             $_SESSION[$state_key] = $out_vars;
  259.         }
  260.     }
  261.  
  262.     /**
  263.      * Reset state in session
  264.      */
  265.      public function reset(&$policy)
  266.      {
  267.          // Start the session if it hasn't been started yet
  268.          if (!session_id()) session_start();
  269.  
  270.          // Unset state
  271.          $state_key = "CONFIDENTCAPTCHA_POLICY_STATE";
  272.          $name_key = "CONFIDENTCAPTCHA_POLICY_NAME";
  273.          if (isset($_SESSION)) {
  274.              unset($_SESSION[$state_key]);
  275.              unset($_SESSION[$name_key]);
  276.          }
  277.      }
  278. }
