  270.     /**
  271.      * Prepare the URL and form parameters for a request
  272.      *
  273.      * @param string  $resource        The resource to call
  274.      * @param string  $method          The HTTP method to use
  275.      * @param array   $params          The parameters to send
  276.      * @param boolean $use_credentials Include API credentials in call
  277.      * @return array with keys 'url' and 'form'
  278.      */
  279.     protected function prep_req($resource, $method, $params,
  280.         $use_credentials)
  281.     {
  282.         $url = $this->captcha_server_url . $resource;
  283.  
  284.         if ($params == NULL) {
  285.             $params = array();
  286.         }
  287.         $params["library_version"] = $this->library_version;
  288.         if ($use_credentials) {
  289.             $params['customer_id']  = $this->customer_id;
  290.             $params['site_id']      = $this->site_id;
  291.             $params['api_password'] = $this->api_password;
  292.             $params['api_username'] = $this->api_username;
  293.         }
  294.  
  295.         $form = NULL;
  296.         if (strtoupper($method) == 'GET') {
  297.             $url .= '?' . http_build_query($params);
  298.         } elseif (strtoupper($method) == 'POST' and $params) {
  299.             $form = http_build_query($params);
  300.         }
  301.         return Array('url'=>$url, 'form'=>$form);
  302.     }

  304.     /**
  305.      * Call the Confident CAPTCHA API
  306.      *
  307.      * @param string  $resource        The resource to call
  308.      * @param string  $method          The HTTP method to use (POST or GET)
  309.      * @param array   $params          The parameters to send
  310.      * @param boolean $use_credentials Include API credentials in call
  311.      * @return CCAP_ApiResponse  The response
  312.      */
  313.     protected function call($resource, $method, $params, $use_credentials)
  314.     {
  315.         $req = $this->prep_req($resource, $method, $params, $use_credentials);
  316.         $url = $req['url'];
  317.         $form = $req['form'];
  318.  
  319.         $ch = curl_init();
  320.  
  321.         if (strtoupper($method) == 'POST') {
  322.             curl_setopt($ch, CURLOPT_POST, TRUE);
  323.  
  324.             if ($form) {
  325.                 curl_setopt($ch, CURLOPT_POSTFIELDS, $form);
  326.             }
  327.         }
  328.  
  329.         curl_setopt($ch, CURLOPT_URL, $url);
  330.  
  331.         curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
  332.         $body = curl_exec($ch);
  333.  
  334.         $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
  335.         if ($body === FALSE) {
  336.             $response = new CCAP_ApiResponse($http_code, curl_error($ch),
  337.                 strtoupper($method), $url, $form, TRUE);
  338.         } else {
  339.             $response = new CCAP_ApiResponse($http_code, $body, 
  340.                 strtoupper($method), $url, $form, TRUE);
  341.         }
  342.  
  343.         curl_close($ch);
  344.         return $response;
  345.     }

  347.     /**
  348.      * Get the user's IP address and User Agent
  349.      *
  350.      * These are used by Confident Technologies to build a risk profile for

  354.     protected function get_user_info()
  355.     {
  356.         if (!empty($_SERVER['HTTP_CLIENT_IP'])) {
  357.             // IP from share internet
  358.             $ip=$_SERVER['HTTP_CLIENT_IP'];
  359.         } elseif (!empty($_SERVER['HTTP_X_FORWARDED_FOR'])) {
  360.             // IP passed from proxy
  361.             $ip=$_SERVER['HTTP_X_FORWARDED_FOR'];
  362.         } elseif (!empty($_SERVER['REMOTE_ADDR'])) {
  363.             // IP directly from caller
  364.             $ip=$_SERVER['REMOTE_ADDR'];
  365.         } else {
  366.             $ip="";
  367.         }
  368.  
  369.         if (!empty($_SERVER['HTTP_USER_AGENT'])) {
  370.             // User-Agent header
  371.             $agent = $_SERVER['HTTP_USER_AGENT'];
  372.         } else {
  373.             $agent = "";
  374.         }
  375.  
  376.         return Array('ip_addr' => $ip, 'user_agent' => $agent);
  377.     }
