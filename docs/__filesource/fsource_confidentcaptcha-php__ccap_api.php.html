  269.     /**
  270.      * Prepare the URL and form parameters for a request
  271.      *
  272.      * @param string  $resource        The resource to call
  273.      * @param string  $method          The HTTP method to use
  274.      * @param array   $params          The parameters to send
  275.      * @param boolean $use_credentials Include API credentials in call
  276.      * @return array with keys 'url' and 'form'
  277.      */
  278.     protected function prep_req($resource, $method, $params,
  279.         $use_credentials)
  280.     {
  281.         $url = $this->captcha_server_url . $resource;
  282.  
  283.         if ($params == NULL) {
  284.             $params = array();
  285.         }
  286.         $params["library_version"] = $this->library_version;
  287.         if ($use_credentials) {
  288.             $params['customer_id']  = $this->customer_id;
  289.             $params['site_id']      = $this->site_id;
  290.             $params['api_password'] = $this->api_password;
  291.             $params['api_username'] = $this->api_username;
  292.         }
  293.  
  294.         $form = NULL;
  295.         if (strtoupper($method) == 'GET') {
  296.             $url .= '?' . http_build_query($params);
  297.         } elseif (strtoupper($method) == 'POST' and $params) {
  298.             $form = http_build_query($params);
  299.         }
  300.         return Array('url'=>$url, 'form'=>$form);
  301.     }

  303.     /**
  304.      * Call the Confident CAPTCHA API
  305.      *
  306.      * @param string  $resource        The resource to call
  307.      * @param string  $method          The HTTP method to use (POST or GET)
  308.      * @param array   $params          The parameters to send
  309.      * @param boolean $use_credentials Include API credentials in call
  310.      * @return CCAP_ApiResponse  The response
  311.      */
  312.     protected function call($resource, $method, $params, $use_credentials)
  313.     {
  314.         $req = $this->prep_req($resource, $method, $params, $use_credentials);
  315.         $url = $req['url'];
  316.         $form = $req['form'];
  317.  
  318.         $ch = curl_init();
  319.  
  320.         if (strtoupper($method) == 'POST') {
  321.             curl_setopt($ch, CURLOPT_POST, TRUE);
  322.  
  323.             if ($form) {
  324.                 curl_setopt($ch, CURLOPT_POSTFIELDS, $form);
  325.             }
  326.         }
  327.  
  328.         curl_setopt($ch, CURLOPT_URL, $url);
  329.  
  330.         curl_setopt($ch, CURLOPT_RETURNTRANSFER, TRUE);
  331.         $body = curl_exec($ch);
  332.  
  333.         $http_code = curl_getinfo($ch, CURLINFO_HTTP_CODE);
  334.         if ($body === FALSE) {
  335.             $response = new CCAP_ApiResponse($http_code, curl_error($ch),
  336.                 strtoupper($method), $url, $form, TRUE);
  337.         } else {
  338.             $response = new CCAP_ApiResponse($http_code, $body, 
  339.                 strtoupper($method), $url, $form, TRUE);
  340.         }
  341.  
  342.         curl_close($ch);
  343.         return $response;
  344.     }

  371.     /**
  372.      * Get the user's IP address and User Agent
  373.      *
  374.      * These are used by Confident Technologies to build a risk profile for

  378.     protected function get_user_info()
  379.     {
  380.         if (!empty($_SERVER['HTTP_CLIENT_IP'])) {
  381.             // IP from share internet
  382.             $ip=$_SERVER['HTTP_CLIENT_IP'];
  383.         } elseif (!empty($_SERVER['HTTP_X_FORWARDED_FOR'])) {
  384.             // IP passed from proxy
  385.             $ip=$_SERVER['HTTP_X_FORWARDED_FOR'];
  386.         } elseif (!empty($_SERVER['REMOTE_ADDR'])) {
  387.             // IP directly from caller
  388.             $ip=$_SERVER['REMOTE_ADDR'];
  389.         } else {
  390.             $ip="";
  391.         }
  392.  
  393.         if (!empty($_SERVER['HTTP_USER_AGENT'])) {
  394.             // User-Agent header
  395.             $agent = $_SERVER['HTTP_USER_AGENT'];
  396.         } else {
  397.             $agent = "";
  398.         }
  399.  
  400.         return Array('ip_addr' => $ip, 'user_agent' => $agent);
  401.     }
