  83. /**
  84.  * Policy interface for the Confident CAPTCHA API
  85.  *
  86.  * In normal siutations, the web server will:
  87.  * - Call the Confident CAPTCHA API to create a CAPTCHA, and inject the
  88.  *   returned HTML into the page,
  89.  * - (Optionally) Use a callback to check a user's CAPTCHA solution with the
  90.  *   Confident CAPTCHA API immediately after they try it, and
  91.  * - At form submission, check with the Confident CAPTCHA API that the user
  92.  *   correctly solved the CAPTCHA.
  93.  *
  94.  * However, if something goes wrong, such as failed API credentials or the
  95.  * site is unavailable due to maintenance, then things get more complicated.
  96.  * This policy interface takes care of the details of failed connections,
  97.  * so that Confident CAPTCHA 'just works'.
  98.  *
  99.  * There are state variables that allow the website developer to handle error
  100.  * cases themselves, as well as customization points so that a derived class
  101.  * can encapsulate the site policy, and different policies can be used on
  102.  * different pages with a one-line change.
  103.  *
  104.  * We don't recommend using this class directly, but instead using one of the
  105.  * provided derived classes, or create your own derived policy.

  107.  * @package confidentcaptcha-php

  277.     /**
  278.      * Check local and remote configuration
  279.      *
  280.      * @param string $callback_url - Callback URL for audio, instant feedback
  281.      * @param array $options - CAPTCHA options
  282.      *
  283.      * @return array with keys 'html' (HTML string) and 'passed' (boolean)
  284.      */
  285.     public function check_config($callback_url = NULL, $options = NULL)
  286.     {
  287.         // Local checks
  288.         $local_config = array(array("Item", "Value",
  289.             "Required Value", "Acceptable?"));
  290.  
  291.         // Check PHP version 5.x
  292.         $php_version = phpversion();
  293.         $php_minimum = "5.0.0";
  294.         if (version_compare($php_version, $php_minimum, '>=')) {
  295.             $php_supported = 'Yes';
  296.         } else {
  297.             $php_supported = 'No';
  298.         }
  299.         $local_config[] = array('PHP version', $php_version, 
  300.             $php_minimum, $php_supported);
  301.  
  302.         // Check cURL extension
  303.         if (extension_loaded('curl')) {
  304.             $curl_version = phpversion('cURL');
  305.             if (empty($curl_version)) $curl_version = '(installed)';
  306.             $curl_supported = 'Yes';
  307.         } else {
  308.             $curl_version = "(not installed)";
  309.             $curl_supported = 'No';
  310.         }
  311.         $local_config[] = array('cURL extension', $curl_version,
  312.             '(installed)', $curl_supported);
  313.         
  314.         // Check SimpleXML extension
  315.         if (extension_loaded('SimpleXML')) {
  316.             $sxml_version = phpversion('SimpleXML');
  317.             if (empty($curl_version)) $sxml_version = '(installed)';
  318.             $sxml_supported = 'Yes';
  319.         } else {
  320.             $sxml_version = "(not installed)";
  321.             $sxml_supported = 'No';
  322.         }
  323.         $local_config[] = array('SimpleXML extension', $sxml_version,
  324.             '(installed)', $sxml_supported);
  325.         
  326.         // Check CAPTCHA API server URL
  327.         $not_set = '(NOT SET)';
  328.         $url = $this->api->captcha_server_url;
  329.         $expected_url = 'http://captcha.confidenttechnologies.com/';
  330.         if ($url == $expected_url) {
  331.             $url_supported = 'Yes';
  332.         } elseif (empty($url)) {
  333.             $url = $not_set;
  334.             $url_supported = 'No';
  335.         } elseif (0 == substr_compare($url, 'http', 0, 4)) {
  336.             $url_supported = 'Maybe';
  337.         } else {
  338.             $url_supported = 'No';
  339.         }
  340.         $local_config[] = array('ccap_server_url', $url, $expected_url,
  341.             $url_supported);
  342.  
  343.         // Check API parameters
  344.         $api_params = array('customer_id', 'site_id', 'api_username', 
  345.             'api_password');
  346.         foreach ($api_params as $param_name) {
  347.             $param = $this->api->$param_name;
  348.             if (empty($param)) {
  349.                 $param = $not_set;
  350.                 $param_ok = 'No';
  351.             } else {
  352.                 $param_ok = 'Yes';
  353.             }
  354.             $local_config[] = array($param_name, $param, '(some value)',
  355.                 $param_ok);
  356.         }
  357.         
  358.         // Check Callback URL
  359.         if (isset($callback_url)) {
  360.             $callback = $callback_url;
  361.         } else {
  362.             $callback = $this->callback_url;
  363.         }
  364.         if (isset($callback)) {
  365.             if (file_exists($callback)) {
  366.                 $callback_ok = 'Yes';
  367.             } else {
  368.                 $callback_ok = 'No';
  369.             }
  370.         } else {
  371.             $callback_ok = 'Somewhat (audio, instant feedback disabled)';
  372.         }
  373.         $local_config[] = array('ccap_callback_url', $callback, 
  374.             '(local path or not set)', $callback_ok);
  375.         
  376.         // Check CAPTCHA options
  377.         $known_options = array(
  378.             'display_style' => '(optional, default "lightbox")',
  379.             'include_audio' => '(optional, default FALSE)',
  380.             'height' => '(optional, default 3)',
  381.             'width' => '(optional, default 3)',
  382.             'length' => '(optional, default 4)',
  383.             'code_color' => '(optional, default "White")'
  384.         );
  385.         foreach ($known_options as $option_name => $expected) {
  386.             if (isset($options[$option_name])) {
  387.                 $option = $options[$option_name];
  388.             } else {
  389.                 $option = $this->$option_name;
  390.             }
  391.             if (is_null($option)) {
  392.                 $option = $not_set;
  393.                 $required = ('(optional' != substr($expected, 0, 9));
  394.                 $option_ok = ($required) ? 'No' : 'Yes';
  395.             } else {
  396.                 // TODO: Should I validate the option?
  397.                 $option_ok = 'Probably';
  398.             }
  399.             $local_config[] = array($option_name, $option, $expected,
  400.                 $option_ok);
  401.         }
  402.  
  403.         # Make local tables
  404.         $local = "<h1>Local Configuration</h1>\n";
  405.         $local .= "<table border=\"1\">\n<tr><th>";
  406.         $head_row = array_shift($local_config);
  407.         $local .= implode('</th><th>', $head_row) . "</th></tr>\n";
  408.         $local_ok = TRUE;
  409.         foreach($local_config as $row) {
  410.             $local .= '<tr><td>'.implode('</td><td>', $row)."</td></tr>\n";
  411.             if (end($row) == 'No') $local_ok = FALSE;
  412.         }
  413.         
  414.         $local .= '</table><br/>';
  415.  
  416.         # Add callback check button
  417.         # TODO: Use javascript to check
  418.         $ok = self::CALLBACK_OK;
  419.         if ($callback_url) {
  420.             $local .= "<form name='callback_check' action='$callback_url'
  421.                     method='post'>
  422.                 <input type='hidden' name='endpoint' value='callback_check' />
  423.                 <input type='submit' value='Click to check the callback' />
  424.             </form>
  425.             <p>
  426.             Response to clicking above should be '$ok'.

  457.     protected function call_api()

  576.     {
  577.         // Get a block_id if needed
  578.         $block_failed = FALSE;
  579.         if (is_null($this->block_id))  {
  580.             $response = $this->create_block();
  581.             if (is_null($this->block_id))  {
  582.                 $block_failed = TRUE;
  583.             }
  584.         }
  585.  
  586.         if (!$block_failed) {
  587.             // Store CAPTCHA creation parameters for future calls
  588.             if (isset($callback_url)) $this->callback_url = $callback_url;
  589.             foreach ($options as $option => $value) {

  592.                     case 'include_audio': $this->include_audio = $value; break;
  593.                     case 'height': $this->height = $value; break;
  594.                     case 'width': $this->width = $value; break;
  595.                     case 'length': $this->length = $value; break;
  596.                     case 'code_color': $this->code_color = $value; break;
  597.                     default: 
  598.                         trigger_error("Unknown CAPTCHA option '$option'",
  599.                             E_USER_WARNING);
  600.                 }
  601.             }
  602.             
  603.             // Can't do audio without a callback
  604.             if ($this->include_audio and !$this->callback_url) {
  605.                 $this->include_audio = FALSE;
  606.             }
  607.         
  608.             // Create the visual CAPTCHA
  609.             $response = $this->call_api('create_visual', $this->block_id,
  610.                 $this->display_style, $this->include_audio, $this->height,
  611.                 $this->width, $this->length, $this->code_color);
  612.         }
  613.         
  614.         // Set visual state
  615.         if ($response->status == 200) {
  616.             $this->visual_creation_succeeded = TRUE;
  617.             $this->visual_authenticated = NULL;
  618.  
  619.             if ($this->callback_url)
  620.             {
  621.                 // HACK: Inject callback URL into code
  622.                 // TODO: Fix API
  623.                 $audio = ($this->include_audio) ? 'true' : 'false';
  624.                 $response->body = str_replace('jQuery(function(', 
  625.                   "var CONFIDENTCAPTCHA_CALLBACK_URL = \"$this->callback_url\";
  626.                    var CONFIDENTCAPTCHA_INCLUDE_AUDIO = $audio;
  627.                    jQuery(function(" , $response->body);
  628.             } else {
  629.                 // HACK: Remove ajax_verify from code
  630.                 // TODO: Fix API
  631.                 $response->body = str_replace('ajax_verify: true,',
  632.                     'ajax_verify: false,', $response->body);
  633.             }
  634.  
  635.             // Find visual_id
  636.             // <input name="..._captcha_id" value='theVisualID'
  637.             // Super complex, but no regex
  638.             $html = $response->body;
  639.             $name_pos = strpos($html, '_captcha_id');
  640.             $v = 'value=';
  641.             $value_pos = strpos($html, $v, $name_pos);
  642.             $quote_char = substr($html, $value_pos + strlen($v), 1);
  643.             $end_value_pos = strpos($html, $quote_char, 
  644.                 ($value_pos + strlen($v) + 1));
  645.             $vid_len = $end_value_pos - $value_pos - strlen($v) - 1;
  646.             $visual_id = substr($html, $value_pos + strlen($v) + 1, $vid_len);
  647.             $this->visual_id = $visual_id;
  648.         } elseif ($response->status == 410) {
  649.             // Too many failures, no more visual CAPTCHAs in block
  650.             $this->block_done = TRUE;
  651.             $this->visual_id = NULL;
  652.         } else {
  653.             // Some other failure - candidate for fail open / closed
  654.             $this->visual_creation_succeeded = FALSE;
  655.             $this->visual_authenticated = FALSE;
  656.             $this->visual_id = NULL;
  657.         }
  658.         
  659.         $this->persist->save($this, 'create_visual', $response);

  661.     }

  702.                 $check_on_server = FALSE;
  703.             }

  705.                 ($block_id != $this->block_id))

  708.             }
  709.         }
  710.  
  711.         // Don't check if we're already authenticated

  720.         
  721.         // Set state

  723.             if ($response->status == 200) {
  724.                 $auth = (strtolower($response->body) == 'true');
  725.                 $this->visual_authenticated = $auth;
  726.             } else {
  727.                 $this->visual_authenticated = FALSE;
  728.             }
  729.         }
  730.         
  731.         $this->persist->save($this, 'check_visual', $response);
  732.         return $this->respond_check_visual($response);
  733.     }

  829.     {

  833.         if ($this->audio_creation_succeeded === FALSE) {
  834.             $check_on_server = FALSE;
  835.         } else {

  839.             {
  840.                 $check_on_server = FALSE;
  841.             }
  842.             if (!is_null($this->block_id) and
  843.                 ($block_id != $this->block_id))
  844.             {
  845.                 $check_on_server = FALSE;
  846.             }
  847.         }
  848.  
  849.         // Don't check if we're already authenticated
  850.         if ($this->audio_authenticated === TRUE) {
  851.             $check_on_server = FALSE;
  852.         }
