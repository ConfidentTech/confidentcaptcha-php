  275.         // Check cURL extension

  280.         } else {
  281.             $curl_version = "(not installed)";
  282.             $curl_supported = 'No';
  283.         }
  284.         $local_config[] = array('cURL extension', $curl_version,
  285.             '(installed)', $curl_supported);
  286.         
  287.         // Check SimpleXML extension
  288.         if (extension_loaded('SimpleXML')) {
  289.             $sxml_version = phpversion('SimpleXML');
  290.             if (empty($curl_version)) $sxml_version = '(installed)';
  291.             $sxml_supported = 'Yes';
  292.         } else {
  293.             $sxml_version = "(not installed)";
  294.             $sxml_supported = 'No';
  295.         }
  296.         $local_config[] = array('SimpleXML extension', $sxml_version,
  297.             '(installed)', $sxml_supported);
  298.         
  299.         // Check CAPTCHA API server URL
  300.         $not_set = '(NOT SET)';
  301.         $url = $this->api->captcha_server_url;
  302.         $expected_url = 'http://captcha.confidenttechnologies.com/';
  303.         if ($url == $expected_url) {
  304.             $url_supported = 'Yes';

  306.             $url = $not_set;
  307.             $url_supported = 'No';
  308.         } elseif ((0 == substr_compare($url, 'http', 0, 4)) and
  309.                   (substr($url, -1) == '/')) {
  310.             $url_supported = 'Maybe';
  311.         } else {
  312.             $url_supported = 'No';
  313.         }
  314.         $local_config[] = array('ccap_server_url', $url, $expected_url,

  316.  
  317.         // Check API parameters

  322.         } else {
  323.             $customer_id_ok = 'Yes';
  324.         }
  325.         $local_config[] = array('customer_id', $customer_id, '(some value)',

  331.             $site_id_ok = 'No';
  332.         } else {
  333.             $site_id_ok = 'Yes';
  334.         }
  335.         $local_config[] = array('site_id', $site_id, '(some value)',
  336.             $site_id_ok);

  338.         $api_username = $this->api->api_username;

  353.             $api_password_ok = 'Yes';

  376.                 $local_ok = FALSE;
  377.         }
  378.         $local .= '</table>';
  379.  
  380.         # Add callback check button
  381.         # TODO: Use javascript to check
  382.         $ok = self::CALLBACK_OK;
  383.         $local .= "<br/>
  384.             <form name='callback_check' action='$callback_url' method='post'>
  385.             <input type='hidden' name='endpoint' value='callback_check' />
  386.             <input type='submit' value='Click to check the callback' />
  387.         </form>
  388.         <p>
  389.         Response to clicking above should be '$ok'.
  390.         </p>
  391.         ";
  392.         $local .= "<br/>\n<h1>Remote Configuration</h1>\n";
  393.         
  394.         // Check credentials with API server
  395.         $response = $this->call_api('check_credentials');
  396.         if ($response->status == 200) {
  397.             $html = $response->body;
  398.             $api_passed = (false === strstr($html, "api_failed='True'"));

  406.             'html' => $local . $html,

  408.         );
  409.         $this->persist->save($this, 'check_config', $response);
  410.         return $response;
  411.     }

  413.     /**
  414.      * Call an API function
  415.      */
  416.     protected function call_api()
  417.     {
  418.         $raw_args = func_get_args();
  419.         $func_name = $raw_args[0];
  420.         unset($raw_args[0]);
  421.         $args = array_values($raw_args);
  422.         $resp = call_user_func_array(array($this->api, $func_name), $args);
  423.         if (($this->api_debug_level >= 2) or
  424.             ($this->api_debug_level == 1 and $resp->status != 200))
  425.         {
  426.             $debug = $this->generate_debug($func_name, $args, $resp);
  427.             $this->handle_debug($debug);
  428.         }
  429.         return $resp;
  430.     }
