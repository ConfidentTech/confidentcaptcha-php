  86. /**
  87.  * Confident CAPTCHA policy for page development
  88.  *
  89.  * Most web developers start by adding Confident CAPTCHA to an existing page
  90.  * in their development environment.  Configuration and usage errors are more
  91.  * common during this phase.  This policy causes these errors to be noisy, so
  92.  * that a developer can quickly fix issues without resorting to logs.
  93.  *
  94.  * @package confidentcaptcha-php
  95.  */
  96. class CCAP_DevelopmentPolicy extends CCAP_Policy
  97. {
  98.  
  99.     /**
  100.      * Storage for API debug messages
  101.      * @var Array 
  102.      */
  103.     var $api_debug_messages = Array();
  104.  
  105.     /**
  106.      * Construct a CCAP_DevelopmentPolicy
  107.      *
  108.      * Sets the 'use_shortcuts' boolean, to avoid bugging the CAPTCHA API
  109.      * server during the awkward development stage.
  110.      *
  111.      * @param CCAP_Api $api The API interface to use for calls
  112.      */
  113.     public function __construct($api, $persist)
  114.     {
  115.         parent::__construct($api, $persist);
  116.         $this->api->use_shortcuts = TRUE;
  117.         $this->api_debug_level = 2;
  118.     }
  119.  
  120.     /**
  121.      * Handle debug information by dumping to output
  122.      */
  123.     protected function handle_debug($debug)
  124.     {
  125.         $debug_html = str_replace("\n", "<br>\n", $debug);
  126.         $this->api_debug_messages[] = $debug_html;
  127.     }
  128.     
  129.     /**
  130.      * Callback for developer policy - add get_debug function
  131.      */
  132.     protected function callback_extensions($endpoint, $request)
  133.     {
  134.         $content = NULL;
  135.         $headers = Array();
  136.         if ($endpoint == 'get_api_debug') {
  137.             $content = array_shift($this->api_debug_messages);
  138.             if (is_null($content)) $content = "";
  139.         } elseif ($endpoint == 'get_policy_dump') {
  140.             $content = htmlentities(var_export($this, TRUE));
  141.         } else {
  142.             $this->api_debug_messages[] = "Unknown callback '$endpoint'";
  143.         }
  144.         if (is_null($content)) {
  145.             $result = NULL;
  146.         } else {
  147.             $result = Array($content, $headers);
  148.         }
  149.         return $result;
  150.     }
  151.     
  152.     /**
  153.      * Get HTML for debug section
  154.      *
  155.      * Inject this into your page to add the API debug messages.

  158.      */
  159.     public function get_debug_html($callback_url)
  160.     {
  161.         // Insertion to get debug messages
  162.         // Don't use on your page
  163.         return <<< DEBUG
  164. <div id="confidentcaptcha_debug" style="display: none">
  165. <h2>Confident CAPTCHA Debug Messages</h2>
  166. <p>
  167. Debug messages will appear here if you are using CCAP_DevelopmentPolicy.
  168. Don't use this debug code in production - it will leak your API credentials.
  169. </p><p>
  170. Use the <a href="#confidentcaptcha_actions">links at the bottom</a> to get
  171. more debug information.
  172. </p>
  173. <ul></ul>
  174. <a name="confidentcaptcha_actions">Actions:</a>
  175. <a href="#" class='confidentcaptcha_debug_refresh'>Fetch new debug
  176.  messages</a>
  177. <a href="#" class='confidentcaptcha_debug_dump'>Dump policy state</a>
  178. </div>
  179. <script type="text/javascript">
  180.     function confidentcaptcha_get_debug(depth, first_call, method)
  181.     {
  182.         if (depth > 5) { return; }
  183.         if (!"$callback_url") {
  184.             $("#confidentcaptcha_debug").css("display","block");
  185.             $("#confidentcaptcha_debug ul").append(
  186.                 "<li>callback_url is not set</li>");
  187.             return;
  188.         }
  189.         $.ajax({
  190.             type: 'POST',
  191.             url: "$callback_url",
  192.             data: {endpoint: method},
  193.             dataType: 'text',
  194.             success: function(html) {
  195.                 $("#confidentcaptcha_debug").css("display","block");
  196.                 if (html) {
  197.                     $("#confidentcaptcha_debug ul").append(
  198.                         "<li>"+html+"</li>");
  199.                     // Recursively call until empty string is returned
  200.                     if (method == 'get_api_debug') {
  201.                         confidentcaptcha_get_debug(depth + 1, false, method);
  202.                     }
  203.                 } else if (depth == 1) {
  204.                     $("#confidentcaptcha_debug ul").append(
  205.                         "<li>No new debug messages</li>");
  206.                 }
  207.             },
  208.             error: function() {
  209.                 if (!first_call) {
  210.                     // Will return 400 if CCAP_DevelopmentPolicy is not used
  211.                     $("#confidentcaptcha_debug ul").append(
  212.                         "<li><b>Error: callback failed.  Are you using" +
  213.                         "CCAP_DevelopmentPolicy?</b></li>"
  214.                     );
  215.                 }
  216.             }
  217.         });
  218.     };
  219.     $(document).ready(function() {
  220.         confidentcaptcha_get_debug(1, true, 'get_api_debug');
  221.  
  222.         $("a.confidentcaptcha_debug_refresh").click(function() {
  223.             confidentcaptcha_get_debug(1, false, 'get_api_debug');

  225.         });
  226.         $("a.confidentcaptcha_debug_dump").click(function() {
  227.             confidentcaptcha_get_debug(1, false, 'get_policy_dump');
  228.             return false;
  229.         });
  230.     });
  231. </script>
  232. DEBUG;
  233.     }
