  87. /**
  88.  * Confident CAPTCHA policy for page development
  89.  *
  90.  * Most web developers start by adding Confident CAPTCHA to an existing page
  91.  * in their development environment.  Configuration and usage errors are more
  92.  * common during this phase.  This policy causes these errors to be noisy, so
  93.  * that a developer can quickly fix issues without resorting to logs.
  94.  *
  95.  * @package confidentcaptcha-php
  96.  */
  97. class CCAP_DevelopmentPolicy extends CCAP_Policy
  98. {
  99.  
  100.     /**
  101.      * Storage for API debug messages
  102.      * @var Array 
  103.      */
  104.     var $api_debug_messages = Array();
  105.  
  106.     /**
  107.      * Construct a CCAP_DevelopmentPolicy
  108.      *
  109.      * Sets the 'use_shortcuts' boolean, to avoid bugging the CAPTCHA API
  110.      * server during the awkward development stage.
  111.      *
  112.      * @param CCAP_Api $api The API interface to use for calls
  113.      */
  114.     public function __construct($api, $persist)
  115.     {
  116.         parent::__construct($api, $persist);
  117.         $this->api->use_shortcuts = TRUE;
  118.         $this->api_debug_level = 2;
  119.     }
  120.  
  121.     /**
  122.      * Handle debug information by dumping to output
  123.      */
  124.     protected function handle_debug($debug)
  125.     {
  126.         $debug_html = str_replace("\n", "<br>\n", $debug);
  127.         $this->api_debug_messages[] = $debug_html;
  128.     }
  129.     
  130.     /**
  131.      * Callback for developer policy - add get_debug function
  132.      */
  133.     protected function callback_extensions($endpoint, $request)
  134.     {
  135.         $content = NULL;
  136.         $headers = Array();
  137.         if ($endpoint == 'get_api_debug') {
  138.             $content = array_shift($this->api_debug_messages);
  139.             if (is_null($content)) $content = "";
  140.         } elseif ($endpoint == 'get_policy_dump') {
  141.             $content = htmlentities(var_export($this, TRUE));
  142.         } else {
  143.             $this->api_debug_messages[] = "Unknown callback '$endpoint'";
  144.         }
  145.         if (is_null($content)) {
  146.             $result = NULL;
  147.         } else {
  148.             $result = Array($content, $headers);
  149.         }
  150.         return $result;
  151.     }
  152.     
  153.     /**
  154.      * Get HTML for debug section
  155.      *
  156.      * Inject this into your page to add the API debug messages.
  157.      * @param string $callback_url The callback URL
  158.      * @return string HTML to inject into page
  159.      */
  160.     public function get_debug_html($callback_url)
  161.     {
  162.         // Insertion to get debug messages
  163.         // Don't use on your page
  164.         return <<< DEBUG
  165. <div id="confidentcaptcha_debug" style="display: none">
  166. <h2>Confident CAPTCHA Debug Messages</h2>
  167. <p>
  168. Debug messages will appear here if you are using CCAP_DevelopmentPolicy.
  169. Don't use this debug code in production - it will leak your API credentials.
  170. </p><p>
  171. Use the <a href="#confidentcaptcha_actions">links at the bottom</a> to get
  172. more debug information.
  173. </p>
  174. <ul></ul>
  175. <a name="confidentcaptcha_actions">Actions:</a>
  176. <a href="#" class='confidentcaptcha_debug_refresh'>Fetch new debug
  177.  messages</a>
  178. <a href="#" class='confidentcaptcha_debug_dump'>Dump policy state</a>
  179. </div>
  180. <script type="text/javascript">
  181.     function confidentcaptcha_get_debug(depth, first_call, method)
  182.     {
  183.         if (depth > 5) { return; }
  184.         if (!"$callback_url") {
  185.             $("#confidentcaptcha_debug").css("display","block");
  186.             $("#confidentcaptcha_debug ul").append(
  187.                 "<li>callback_url is not set</li>");
  188.             return;
  189.         }
  190.         $.ajax({
  191.             type: 'POST',
  192.             url: "$callback_url",
  193.             data: {endpoint: method},
  194.             dataType: 'text',
  195.             success: function(html) {
  196.                 $("#confidentcaptcha_debug").css("display","block");
  197.                 if (html) {
  198.                     $("#confidentcaptcha_debug ul").append(
  199.                         "<li>"+html+"</li>");
  200.                     // Recursively call until empty string is returned
  201.                     if (method == 'get_api_debug') {
  202.                         confidentcaptcha_get_debug(depth + 1, false, method);
  203.                     }
  204.                 } else if (depth == 1) {
  205.                     $("#confidentcaptcha_debug ul").append(
  206.                         "<li>No new debug messages</li>");
  207.                 }
  208.             },
  209.             error: function() {
  210.                 if (!first_call) {
  211.                     // Will return 400 if CCAP_DevelopmentPolicy is not used
  212.                     $("#confidentcaptcha_debug ul").append(
  213.                         "<li><b>Error: callback failed.  Are you using" +
  214.                         "CCAP_DevelopmentPolicy?</b></li>"
  215.                     );
  216.                 }
  217.             }
  218.         });
  219.     };
  220.     $(document).ready(function() {
  221.         confidentcaptcha_get_debug(1, true, 'get_api_debug');
  222.  
  223.         $("a.confidentcaptcha_debug_refresh").click(function() {
  224.             confidentcaptcha_get_debug(1, false, 'get_api_debug');
  225.             return false;
  226.         });
  227.         $("a.confidentcaptcha_debug_dump").click(function() {
  228.             confidentcaptcha_get_debug(1, false, 'get_policy_dump');
  229.             return false;
  230.         });
  231.     });
  232. </script>
  233. DEBUG;
  234.     }
